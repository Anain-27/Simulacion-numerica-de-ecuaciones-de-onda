\section{ Conceptos previos}
\subsection{ Espacios de Lebesgue}
\begin{frame}
	
	\textcolor{blue}{\Large{Espacios de Lebesgue}}
	\vspace{0.5cm}
	
	\begin{block}{Definición: Espacio de Lebesgue}
		\footnotesize{	Para $E\subset \mathbb{R}^n$, con $n\in\mathbb{N}$,
			medible y $p\in \mathbb{R}$ tal que  $1\leq p < \infty$, definimos el Espacio de Lebesgue $L^p(E)$, como la colección de funciones medibles, en la que se realiza una identificación entre las funciones que son iguales en casi todo, que son $p$-integrables, es decir, las funciones $f$ para las cuales 
			\begin{equation*}
				||f||_{L^p(E)}=\left(\int_{E}|f|^p\right)^{\frac{1}{p}}<\infty.
			\end{equation*}}
	\end{block}
	
	\vspace{0.75cm}
	\pause
	
	\begin{block}{Definición: Espacio de Banach}
		\footnotesize{Un espacio de Banach $(E,||\cdot||)$ es un espacio vectorial normado, en el que $E$ es completo bajo la métrica inducida por la norma $||\cdot||$. Es decir, cada serie de Cauchy en $E$ es una serie convergente.}
	\end{block}
	
\end{frame}
\begin{frame}

	\begin{block}{Definición: Producto escalar o interno}
		\footnotesize{Sea $E$ un espacio vectorial. Una función $(\cdot,\cdot):E\times E\rightarrow \mathbb{R} $ se llamará producto escalar o interno, si para todo $u,v,w\in E$ y  $\alpha, \beta \in \mathbb{R}$ escalares, se satisfacen las siguientes propiedades:
		\begin{itemize}
			\item Linealidad: $ (\alpha u+\beta v,w)= \alpha(u,w)+\beta(v,w)$ .
			\item Simetría: $ (u,v)=(v,u)$ .
			\item Positividad: $ (u,u)\geq 0$ , y $ (u,u)=0$  si y solo si $u=0$ .
			
		\end{itemize}
		}
	\end{block}
	
	\vspace{0.75cm}
	\pause
	
	\begin{block}{Definición: Espacio de Hilbert}
		\footnotesize{Un espacio de Banach $E$, se llama de Hilbert, si tiene un producto escalar asociado y es completo bajo su métrica.}
	\end{block}
	
\end{frame}



\subsection{Producto escalar en $L^2$}
\begin{frame}
	
	\textcolor{blue}{\Large{Producto escalar en $L^2$}}
	\vspace{0.5cm}
	
	\begin{block}{Definición: Producto escalar en $L^2$}
		\footnotesize{Sean $f$ y $g$ dos funciones definidas en $E$, definimos el producto escalar sobre $L^2(E)$ de ellas como:
			$$\left< f,g\right>_{L^2(E)}= \int_{E}f(x)g(x)dx.$$
		}
	\end{block}
	\vspace{1cm}
	\pause
	Para funciones que dependen de más de una variable, como por ejemplo del espacio y el tiempo, $u(x,t)$:
	
	\begin{equation*}
		\begin{split}
			\langle u_1(\cdot,t),u_2(\cdot,t)\rangle_{L^2(E)}&= \int_{E}u_1(x,t)u_2(x,t)dx,\\
			||u(\cdot,t)||_{L^2(E)}& = \sqrt{\left< u(x,t),u(x,t)\right>_{L^2(E)}}.
		\end{split}	
	\end{equation*}
	
\end{frame}



\begin{frame}

	\begin{block}{Definición:Producto escalar en dominios discretos}
			\footnotesize{Sean $u$ y $w$ dos funciones definidas en nuestro mallado y sea $D=\{ x_1,\dots,x_{m-1}\}$. 
			Fijando un instante temporal $t_r$, definimos el producto escalar sobre $\mathbb{R}^{m-1}$ como:
			\begin{equation*}
				\left(u(\cdot,t_r),w(\cdot,t_r)\right)_{D}= \sum_{j=1}^{m-1}h u(x_j,t_r)w(x_j,t_r),
			\end{equation*}
			Para este tipo de casos usaremos la siguiente notación:
			
			\begin{equation}\label{eq:notacioninteriordiscretomalla}
				\left(u_{\cdot,r},w_{\cdot,r}\right)_{D}= \sum_{j=1}^{m-1}h u_{j,r}w_{j,r}.
			\end{equation}
			
			La norma asociada a este producto escalar es:
			$||u||_{D}= \sqrt{\left( u,u\right)_D}\geq 0$.}
	\end{block}
	
\end{frame}



\section{Definición de la ecuación de onda}
\begin{frame}
	
	\textcolor{blue}{\Large{Definición de la ecuación de onda}}
	\vspace{0.5cm}
	
	Denotamos a la función $u(x,t)$ como $u$.
	\begin{equation*}
		u_{tt}- c^2 \Delta u = f ,\quad \Omega \times[0,+\infty),
	\end{equation*}
	donde $\Omega \subset \mathbb{R}^n$, $f$ es una función de valores reales, $c>0$ es una constante de propagación y el operador 
	\begin{equation*}
		\Delta u = \sum_{j=1}^{N}\frac{\partial^2u}{\partial x^2_j},
	\end{equation*}
	es conocido como el laplaciano.
\end{frame}

\begin{frame}
	
	\textcolor{blue}{\Large{Clasificación de las EDP}}\\
	\vspace{0.5cm}
	
	Una ecuación lineal de segundo orden en forma estándar sería como sigue:
	\begin{equation*}
		Au_{tt}+Bu_{tx}+Cu_{xx}+Du_{t}+Eu_{x}+Fu= G,
	\end{equation*}
	
	\begin{block}{Definición: Clasificación de las EDP de segundo orden}	
		
		\footnotesize{Si en todos los puntos $(x,t)$ de una región $W\subset \mathbb{R}^2$ se cumple que:
		\begin{itemize}
			\item $B^2(x,t)-4A(x,t)C(x,t)>0$, entonces la EDP se dice hiperbólica.
			\item $B^2(x,t)-4A(x,t)C(x,t)=0$, entonces la EDP se dice parabólica.
			\item $B^2(x,t)-4A(x,t)C(x,t)<0$, entonces la EDP se dice elíptica.
			
		\end{itemize}}
	\end{block}
	\pause
	Para la ecuación de onda, tenemos $B^2-4AC=4c^2>0$.\\
	La ecuación es hiperbólica.
\end{frame}



\subsection{Soluciones de tipo onda viajera}

\begin{frame}
	
	\textcolor{blue}{\Large{Soluciones de tipo onda viajera}}
	\vspace{0.5cm}
	
	La ecuación admite  la siguiente solución de tipo onda viajera:
	\begin{equation}\label{ondaviajera}
		u(x,t)=p_+(x-ct)+p_-(x+ct),
	\end{equation}
	
	\pause
	Utilizando la condición $u(x,0)=f(x),$ para  $x\in (0,L)$,
	\begin{equation*}
		f(x)=u(x,0)=p_+(x)+p_-(x),
	\end{equation*}
	derivando respecto a $x$,
	\begin{equation}\label{f}
		f'(x)=p_{+\ x}(x)+p_{-\ x}(x).
	\end{equation}
\end{frame}

\begin{frame}
	La ecuación admite  la siguiente solución de tipo onda viajera:
	\begin{equation}\label{ondaviajera}
		u(x,t)=p_+(x-ct)+p_-(x+ct),
	\end{equation}
	
	\pause
	
	Derivamos la ecuación \eqref{ondaviajera}, 
	\begin{equation*}
		u_t=\frac{\partial}{\partial t}(p_+(x-ct)+p_-(x+ct))=cp_+'(x-ct)-cp_-'(x+ct),
	\end{equation*}
	sustituyendo la condición $	u_t(x,0)=g(x)$ para  $x\in (0,L)$, 
	\begin{equation}\label{g}
		g(x)=u_t(x,0)=-cp_+(x)+cp_-(x),
	\end{equation}
\end{frame}


\begin{frame}
	De las igualdades \eqref{f} y \eqref{g} se sigue:
	$$p_-'(x)= \frac{1}{2}f'(x) + \frac{1}{2c}g(x),\quad \textbf{y}\quad p_+'(x)= \frac{1}{2}f'(x) - \frac{1}{2c}g(x).$$
	
	\pause
	Integrando entre $0$ y $x+ct$ y $0$ y $x-ct$ respectivamente,
	\begin{equation*}
		p_-(x+ct)-p_-(0)=\frac{1}{2}\left(f(x+ct)-f(0)\right) + \frac{1}{2c}\int_{0}^{x+ct}g(s) ds +A,
	\end{equation*}
	\begin{equation*}
		p_+(x-ct)-p_-(0)=\frac{1}{2}\left(f(x-ct)-f(0)\right) - \frac{1}{2c}\int_{0}^{x-ct}g(s) ds +B.
	\end{equation*}
	
\end{frame}

\begin{frame}
	\begin{equation*}
		p_-(x+ct)=\frac{1}{2}f(x+ct) + \frac{1}{2c}\int_{0}^{x+ct}g(s) ds,
	\end{equation*}
	\begin{equation*}
		p_+(x-ct)=\frac{1}{2}f(x-ct) - \frac{1}{2c}\int_{0}^{x-ct}g(s) ds,
	\end{equation*}
	\pause
	La solución del problema de contorno es la siguiente:
	\begin{equation}\label{eq:uviajera}
		u(x,t)= \frac{1}{2}\left(f(x+ct)+f(x-ct)\right) + \frac{1}{2c}\int_{x-ct}^{x+ct}g(s)
		ds. 
	\end{equation}
\end{frame}




\subsection{La energía en la ecuación de onda}
\begin{frame}
	
	\textcolor{blue}{\Large{La energía en la ecuación de onda}}
	\vspace{0.5cm}
	\begin{equation}\label{ondaut}
		\left<u_t,u_{tt}\right>_{L^2(\Omega)} = c^2\left<u_{t},u_{xx}\right>_{L^2(\Omega)}.
	\end{equation} 
	
	Tendremos que ponerlo de la forma $\frac{\partial}{\partial t}\left(\cdot\right)$.
	\pause
	\begin{equation}\label{eq:sutituyendo}
		\begin{split}
			0&=\left<u_t,u_{tt}\right>_{L^2(\Omega)} -c^2\left<u_{t},u_{xx}\right>_{L^2(\Omega)}\\
			&=\int_0^L u_{tt}u_{t} dx + c^2\left(\int_0^L u_{tx}u_{x} dx-\left.u_tu_x\right|_0^L\right)
		\end{split}
	\end{equation}
	\pause
	Utilizando las condiciones de contorno:
	$$\left.u_tu_x\right|_0^L= u_t(\cdot,L)u_x(\cdot,L) - u_t(\cdot,0)u_x(\cdot,0)=0. $$
	
\end{frame}

\begin{frame}
	Usando la regla de la cadena obtenemos:
	\begin{equation*}
		0=\frac{\partial}{\partial t}\left( \frac{1}{2}||u_t||_{L^2(\Omega)}^2+\frac{c^2}{2}||u_x||_{L^2(\Omega)}^2\right)
	\end{equation*} 


	\pause
	Denotamos:
	$$E(u)=\frac{1}{2}||u_t||_{L^2(\Omega)}^2+\frac{c^2}{2}||u_x||_{L^2(\Omega)}^2,$$
	
	\pause

	$$\frac{\partial}{\partial t}E(u)=0.$$ 
	Satisface la ley de conservación de la energía.
\end{frame}



\section{Resolución numérica de la ecuación de onda}
\subsection{Método de las diferencias finitas}
\begin{frame}

	\textcolor{blue}{\Large{Mallado}}
	\vspace{0.5cm}
	
	\begin{block}{Definición: Mallado}
		\footnotesize{Sea $[a,b]\times[c,d]$ con $b\ge a,d\ge c$ y sean $n,m\in \mathbb{N}$. Definimos $h$ como tamaño del mallado en espacio y el paso de tiempo $k$, de forma que los puntos de nuestro mallado serán:
			$$(x,t)_{j.r}=(x_j,t_r), \ \ \  \text{donde}$$
			$$\begin{array}{lcc}   
				x_j=jh & para &  j=0,1,\dots,m,\\
				t_r = rk & para & r=0,1,\dots,n.\\
			\end{array} $$
		}
	\end{block}
	\pause
	\begin{figure}[!ht]
		\centering
		\includegraphics[scale=0.48]{mallado.png}
		\caption{Mallado}
	\end{figure}
	
\end{frame}

\begin{frame}
	
	Definimos el mallado:
	$$x_j=jh \ \ \ \  \text{para} \ \ \ \  j=0,\dots,m,$$
	con $x_0=0$, $x_m=L$.\\
	\pause
	
	$$t_r = rk \ \ \ \  \text{para} \ \ \ \  r=0,1,\dots,n,$$
	con $t_0=0$, $t_n=T$.\\
	\pause
	La ecuación evaluada en los puntos del mallado sería:
	
	\begin{equation*}
		u_{tt}(x_j,t_r)=c^2u_{xx}(x_j,t_r).
	\end{equation*}

\end{frame}

\begin{frame}
	
	\textcolor{blue}{\Large{Método de las diferencias finitas}}
	\vspace{0.5cm}
	
	Aproximaremos la solución $u$, en estos puntos, llamándola  $u(x_j,t_r)\approx u_{j,r}$.
	
	Utilizando una aproximación centrada :
	
	\begin{equation*}
		u_{xx}(x_j,t_r) \approx \frac{u_{j+1,r} - 2u_{j,r} +u_{j-1,r}}{h^2},
	\end{equation*}
	
	\begin{equation*}
		u_{tt}(x_j,t_r) \approx \frac{u_{j,r+1} - 2u_{j,r} +u_{j,r-1}}{k^2}.
	\end{equation*}
	
	\pause
	\begin{figure}[!ht]
		\centering
		\includegraphics[scale=0.45]{UsadosPaso.png}
	\end{figure}
	
\end{frame}

\begin{frame}
	
	Definiendo $\mu  = \frac{ck}{h}$.
	
	$$ u_{j,r+1} = \mu ^2( u_{j+1,r}+u_{j-1,r})  + 2(1-\mu^2)  u_{j,r} -  u_{j,r-1}.$$
	
	\pause
	
	Añadiendo las condiciones de contorno:
	
	\begin{equation*}
		\begin{split}
			u_{j,r+1} &= \mu ^2( u_{j+1,r}+u_{j-1,r})  + 2(1-\mu^2)  u_{j,r} -u_{j,r-1},\\
			&\text{para} \ j=1,\dots,m-1 \text{ y }r=1,\dots,n-1,
		\end{split}
	\end{equation*} 
	\begin{equation*}
		\begin{array}{lc}   
			u_{j,0}=f(x_j), &   j=0,1,\dots,m-1,
		\end{array}
	\end{equation*} 
	\begin{equation*}
		\begin{array}{lc}   
			u_{j,1}=u_{j,0}+k g(x_j), &  j=0,1,\dots,m-1,
		\end{array}
	\end{equation*} 
	\begin{equation*}
		\begin{array}{lc}   
			u_{0,r}=u_{m,r}=0, &  r=0,1,\dots,n-1.
		\end{array}
	\end{equation*} 
\end{frame}

\subsection{Consistencia}
\begin{frame}
	
	\textcolor{blue}{\Large{Consistencia}}
	\vspace{0.5cm}
	\begin{equation*}
		\underbrace{\frac{u(j,r+1)-2u(j,r)+u(j,r-1)}{k^2}}_{(1)}-c^2\underbrace{\frac{u(j+1,r)-2u(j,r)+u(j-1,r)}{h^2}}_{(2)}=0.
	\end{equation*}
	\pause
	Desarrollando las series de Taylor en (1) y (2):
	\begin{equation*}
		\begin{split}
			(1)=&\frac{u(j,r+1)-2u(j,r)+u(j,r-1)}{k^2}= u_{tt}+2\frac{k^2 u_{4t}}{4!}+2\frac{k^4 u_{6t}}{6!}+O(h^6),\\
			(2)=&\frac{u(j+1,r)-2u(j,r)+u(j-1,r)}{h^2}=u_{xx}+2\frac{h^2 u_{4x}}{4!}+2\frac{h^4 u_{6x}}{6!}+O(k^6).
		\end{split}
	\end{equation*}
	\pause
	\begin{equation*}
		\begin{split}
			T_{h,k}u\mathrel{:=}&u_{tt}+2\frac{k^2 u_{4t}}{4!}+2\frac{k^4 u_{6t}}{6!}-c^2u_{xx}-2c^2 \frac{h^2 u_{4x}}{4!}-2c^2\frac{h^4 u_{6x}}{6!}\dots=\\
			=&u_{tt}+2\frac{k^2 u_{4t}}{4!}-c^2u_{xx}-2c^2\frac{h^2 u_{4x}}{4!}+O(k^4)+O(h^4).
		\end{split}
	\end{equation*}
\end{frame}



\begin{frame}
	De la solución exacta obtenemos $u_{4x}=\frac{u_{4t}}{c^4}$ y por tanto:
	
	\begin{equation*}
		\begin{split}
			T_{h,k}u=&u_{tt}+2\frac{k^2 u_{4t}}{4!}-c^2u_{xx}-2c^2\frac{h^2 u_{4x}}{4!}+O(k^4)+O(h^4) = \\
			=&2\frac{k^2 u_{4t}}{4!}-2c^2\frac{h^2 u_{4t}}{4!c^4}+O(k^4)+O(h^4)=\\
			=& \frac{2}{4!}(k^2-\frac{h^2}{c^2})u_{4t}+O(k^4)+O(h^4)=\\
			=& O(k^2)+O(h^2).		
		\end{split}
	\end{equation*}
	\pause
	
	Así hemos comprobado que el esquema tiene orden de consistencia dos en espacio y en tiempo.
\end{frame}




\subsection{Análisis de Von Neumann}

\begin{frame}
	
	\textcolor{blue}{\Large{Análisis de Von Neumann}}
	\vspace{0.5cm}
	
	Relacionamos cada aproximación numérica del esquema con la solución exacta y el error del mismo:
	\begin{equation*}
		u_{j,r}=u(x_j,t_r)+\varepsilon_{j,r}
	\end{equation*}
	\pause
	Sustituyéndolo en el sistema:
	\begin{equation*}
		\begin{split}
			&\Bigl(\frac{u(x_j,t_{r+1}) - 2u(x_{j},t_{r}) +u(x_j,t_{r-1})}{k^2}\\
			&- c^2 \frac{u(x_{j+1},t_{r})- 2u(x_{j},t_{r}) +u(x_{j-1},t_{r})}{h^2}\Bigr)+\\
			&+\left(\frac{\varepsilon_{j,r+1} - 2\varepsilon_{j,r} +\varepsilon_{j,r-1}}{k^2} - c^2 \frac{\varepsilon_{j+1,r} - 2\varepsilon_{j,r} +\varepsilon_{j-1,r}}{h^2}\right)=0.
		\end{split}
	\end{equation*}
\end{frame}


\begin{frame}
	Utilizando el error de tipo onda $\varepsilon_{j,r}=g^re^{ij\beta h}$:
	\pause
	\begin{equation*}
		g^{r+1}e^{ij\beta h} = \mu ^2( g^{r}e^{i(j+1)\beta h}+g^{r}e^{i(j-1)\beta h})  +2(1-\mu^2)  g^{r}e^{ij\beta h} -  g^{r-1}e^{ij\beta h}.
	\end{equation*}
	 \pause
	 
	 Realizando manipulaciones obtenemos y denotando $A=\left(1-2\mu ^2sin^2\left(\frac{\beta h}{2}\right)\right)$:
	 \begin{equation*}
		\begin{split}
			g= &\frac{2A\pm \sqrt{(-2A)^2-4}}{2}=A\pm \sqrt{A^2-1}=\\
			= &1-2\mu ^2sin^2\left(\frac{\beta h}{2}\right) \pm \sqrt{\left(1-2\mu ^2sin^2\left(\frac{\beta h}{2}\right)\right)^2-1}.
		\end{split}
	\end{equation*}
	
\end{frame}

\begin{frame}
	Vamos a estudiar cuando $|g|\leq1$:
	
	\begin{itemize}
		\item Si $A<-1$:\\ La solución $g = A-\sqrt{A^2-1}<-1$ y por lo tanto siempre se da $|g|>1$ y el esquema será en cualquier caso inestable.
		
		\pause
		
		\item Si $ -1\leq A\leq 1$:\\ 
		En este caso se tendrá que $g$ es un número complejo, ya que $-1\leq A^2-1\leq0$. 
		$$g=A\pm \sqrt{A^2-1}=A\pm \sqrt{1-A^2}i.$$
		
		Por lo tanto, tomando módulo, ambas soluciones quedarían,
		$$|g|= \sqrt{A^2+\sqrt{1-A^2}^2}=A^2+1-A^2=1, $$
		luego nuestro esquema sería estable.
		
	\end{itemize}
\end{frame}

\begin{frame}
	Veamos que condición se debe imponer sobre $\mu$ para que se tenga $$ -1\leq A\leq 1,$$ 
	\pause
	Centrándonos en $sin^2\left(\frac{\beta h}{2}\right)$, con esto tenemos
	$$0\leq sin^2\left(\frac{\beta h}{2}\right)\leq 1,$$
	\pause
	$$1\geq 1-2\mu^2sin^2\left(\frac{\beta h}{2}\right)\geq 1-2\mu^2.$$
	
	
	Ahora solo necesitamos que $1-2\mu^2\geq -1$, para lo cual debe ser
	$$2\mu^2\leq2,$$
	\pause
	\begin{equation}\label{mu}
		\mu\leq 1, \textbf{ (Condición CFL)}
	\end{equation}
\end{frame}



\subsection{Resolución numérica para soluciones de tipo viajero}
\begin{frame}
	
	\textcolor{blue}{\Large{Resolución numérica para soluciones de tipo viajero}}
	\vspace{0.5cm}
	\begin{equation*}
		\begin{split}
			w^+(x_j,t_r)&=p_+(x_j-ct_r),\\
			w^-(x_j,t_r)&=p_-(x_j+ct_r).
		\end{split}
	\end{equation*} 
	
	Ahora tendremos que la aproximación de la solución de $u$ será,
	\begin{equation}\label{eq:aproximacionw}
		u_{j,r}=w^+_{j,r}+w^-_{j,r}.
	\end{equation}
	
	\pause
	\begin{equation*}
		\begin{split}
			p_-(x_j+ct_r)&=\frac{1}{2}f(x_j+ct_r) + \frac{1}{2c}\int_{0}^{x_j+ct_r}g(s)\partial s =\\
			&= \frac{1}{2}f(x_j+ct_r) + \frac{1}{2c}\int_{0}^{x_j}g(s)d s + \frac{1}{2c}\int_{x_j}^{x_j+ct_r}g(s)d s.
		\end{split}
	\end{equation*}
\end{frame}

\begin{frame}
	Usando la regla del trapecio:
	\begin{equation*}
		\begin{split}
			w^-_{j,r}&= \frac{1}{2}f(x_j+ct_r) + \frac{h}{4c}\left(g(x_0)+2\sum_{i=1}^{j-1}g(x_i)+ g(x_j)\right)\\
			& + \frac{ck}{4c}\left(g(x_j)+2\sum_{i=1}^{r-1}g(x_j+ct_i)+ g(x_j+ct_r)\right).
		\end{split}
	\end{equation*}
	\pause
	
	Análogamente para $p_+(x_j-ct_r)$
	\begin{equation*}
		\begin{split}
			w^+_{j,r}&= \frac{1}{2}f(x_j+ct_r) + \frac{h}{4c}\left(g(x_0)+2\sum_{i=1}^{j-1}g(x_i)+ g(x_j)\right)\\
			& - \frac{ck}{4c}\left(g(x_j-ct_r)+2\sum_{i=r-1}^{1}g(x_j-ct_i)+ g(x_j)\right).
		\end{split}
	\end{equation*}
\end{frame}



\subsection{Ley de energía discreta}
\begin{frame}
	
	\textcolor{blue}{\Large{Ley de energía discreta}}\\
	\vspace{0.5cm}
	En caso continuo: 
	\begin{equation*}
		\left<u_t(\cdot,t),u_{tt}(\cdot,t)\right>_{L^2(\Omega)} = c^2\left<u_{t}(\cdot,t),u_{xx}(\cdot,t)\right>_{L^2(\Omega)}.
	\end{equation*} 
	
	\pause
	\begin{block}{Definición: Aproximaciones progresivas$\text{,}$ regresivas y centradas}
		
		Sea $u$ una aproximación de la solución definida en un mallado.\\ Definiremos las diferencias aproximadas hacia adelante, atrás o centradas respectivamente, como:
		\begin{itemize}
			\item $\delta_{x^+}u_{j,r}= \frac{1}{h}\left(u_{j+1,r}-u_{j,r}\right)$, 
			\item $\delta_{x^-}u_{j,r}=\frac{1}{h}\left(u_{j,r}-u_{j-1,r}\right)$, 
			\item $\delta_{x}u_{j,r}= \frac{1}{2h}\left(u_{j+1,r}-u_{j-1,r}\right)$. 
		\end{itemize}
		Podemos hacer lo mismo para el eje temporal, denotándolas como $\delta_{t^+},\delta_{t^-},\delta_{t}$
	\end{block}
	
	
\end{frame}


\begin{frame}
	\begin{block}{Definición: Aproximación centrada de la derivada de segundo orden}
		Sea $u$ una aproximación de la solución definida en un mallado. Definimos la aproximación centrada de la derivada de segundo orden como:
		
		\begin{equation*}
			\begin{split}
				\delta_{xx}u_{j,r}&=\frac{1}{h^2}\left(e_{x^+}u_{j,r}-2u_{j,r}+e_{x^-}u_{j,r}\right)=\\
				&=\frac{1}{h^2}\left(u_{j+1,r}-2u_{j,r}+u_{j-1,r}\right)
			\end{split}
		\end{equation*} 
		Al igual que la definición anterior esto será análogo en el eje temporal.
	\end{block}
	
	\pause
	\vspace{0.5cm}
	En caso discreto, tomando $D=\{x_0,x_1,\dots, x_{m-1}\}$:
	$$\left(\delta_{tt}u,\delta_tu\right)_D= c^2 \left(\delta_{xx}u,\delta_tu\right)_D. $$
\end{frame}

\begin{frame}
	De la definición anterior:
	\begin{equation*}
		\begin{split}
			\left(\delta_{xx}u_{\cdot,r},\delta_t u_{\cdot,r}\right)_D &= \left(\delta_{x^+x^-}u_{\cdot,r},\delta_t u_{\cdot,r}\right)_D,\\
		=-\left(\delta_{x^-}u_{\cdot,r},\delta_{x^-}\delta_{t} u_{\cdot,r}\right)_D - (\delta_{t} &u_{0,r})(\delta_{x^+}u_{-1,r})+ (\delta_{t} u_{m,r})(\delta_{x^+}u_{m-1,r}).
		\end{split}
	\end{equation*}
	
	\pause
	
	Estos últimos términos serán $0$, ya que, 
	$$\delta_{t} u_{0,r}= \frac{1}{k}\left(u_{0,r+1}-u_{0,r-1}\right)=0,$$
	
	$$\delta_{t} u_{m,r}= \frac{1}{k}\left(u_{m,r+1}-u_{m,r-1}\right)=0.$$
	
	\pause
	
	El funcional de energía discreto asociado a nuestro esquema conserva la energía.
	$$\delta_{t}E(u_{j,r})= \delta_t\left(||\delta_tu_{\cdot.r}||_D^2+||\delta_{x^-}u_{\cdot.r}||_D^2\right)=0.$$
\end{frame}
